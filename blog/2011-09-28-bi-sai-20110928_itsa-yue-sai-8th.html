int Check(int x, int y, int Num)
{
    int a, b;
    for(a=0;a<N*N;a++)
        if(Map[x][a] == Num && y!=a)    return 0;
    for(a=0;a<N*N;a++)
        if(Map[a][y] == Num && x!=a)    return 0;
    int xx=x/N*N, yy=y/N*N;
    for(a=0;a<N;a++)
        for(b=0;b<N;b++)
            if(Map[xx+a][yy+b] == Num && (xx+a!=x && yy+b!=y))    return 0;
    return 1;
}

        return;
